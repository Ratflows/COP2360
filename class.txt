// Base class representing a generic person 
public class Person 
{ 
        public string Name { get; set; } 
        public int Age { get; set; } 
public Person() { } 
public Person(string name, int age)
{ 
Name = name; 
Age = age; 
} 
public virtual string GetInfo() 
{ 
return $"Name: {Name}, Age: {Age}"; 
} 
} 
// Derived class representing a student 
public class Student : Person 
{ 
public string Major { get; set; } 
public double GPA { get; set; } 
public Student() { } 

public Student(string name, int age, string major, double gpa) 
: base(name, age) 
{ 
Major = major; 
GPA = gpa; 
} 
public override string GetInfo() 
{ 
return base.GetInfo() + $", Major: {Major}, GPA: {GPA}"; 
} 
} 
// Derived class representing an employee 
public class Employee : Person 
{ 
public string Department { get; set; } 
public double Salary { get; set; } 
public Employee() { } 
public Employee(string name, int age, string department, double salary) : base(name, age) 
{ 
Department = department; 
Salary = salary; 
} 
public override string GetInfo() 
{ 
return base.GetInfo() + $", Department: {Department}, Salary: {Salary:C}"; 
}
}
